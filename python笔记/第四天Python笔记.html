def sum_5(a,*tuple_3):
    number =a
    for x in tuple_3:
        number+=x
    #把number作为结果返回
    #功能1：返回结果
    return number

#匿名函数
#lambda关键字创建匿名函数
#lambda参数列表：函数返回结果（运算公式）

#调用匿名函数
result_number=sum_9(10,20)
print result_number
